99  void CommonSocket::listen_on_port(Port port)
100  {
101  if (this->_binded) throw SocketException("[listen_on_port] Socket already binded to a port, close the socket before to re-bind");
102 
103  if (!this->_opened) this->open();
104 
105  Address address(port);
106 
107  if (bind(this->_socket_id, (struct sockaddr*)&address, sizeof(struct sockaddr)) == -1)
108  {
109  stringstream error;
110  error << "[listen_on_port] with [port=" << port << "] Cannot bind socket";
111  throw SocketException(error.str());
112  }
113 
114  this->_binded = true;
