101  {

103  if (!this->_opened) throw SocketException("[send] Socket not opened");
104 
105  len *= sizeof(T);

137  void TCP::send_file(string file_name)
138  {
139  unsigned long long file_size;
140  char chunk[SOCKET_MAX_BUFFER_LEN];
141  char sync;
142  fstream fp(file_name.c_str(), ios::in | ios::binary);
143 
144  if (!fp.is_open())
145  {
146  stringstream error;
147  error << "[send_file] with [filename=" << file_name << "] Cannot open the file";
148  throw SocketException(error.str());
149  }
150 
151  fp.seekg(0, ifstream::end);
152  file_size = fp.tellg();
153  fp.seekg(0, ifstream::beg);
154  this->send<unsigned long long>(&file_size, 1);
155 
156  for (unsigned long long i = 0; i < file_size / SOCKET_MAX_BUFFER_LEN; i++)
157  {
158  this->receive<char>(&sync, 1);
159  fp.read(chunk, SOCKET_MAX_BUFFER_LEN);
160  this->send<char>(chunk, SOCKET_MAX_BUFFER_LEN);
161  }
162 
163  if ((file_size % SOCKET_MAX_BUFFER_LEN) != 0)
164  {
165  this->receive<char>(&sync, 1);
166  fp.read(chunk, file_size % SOCKET_MAX_BUFFER_LEN);
167  this->send<char>(chunk, file_size % SOCKET_MAX_BUFFER_LEN);
168  }
169 
170  fp.close();
171  }
