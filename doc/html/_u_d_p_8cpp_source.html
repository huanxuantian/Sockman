113  template <class T>
114  int UDP::receive(Address *address, T *data, size_t len, unsigned int *received_elements)
115  {
116  if (!this->_opened) this->open();
117  if (!this->_binded) throw SocketException("[receive] Make the socket listening before receiving");
118 
119  len *= sizeof(T);
120  if (len > (SOCKET_MAX_BUFFER_LEN * sizeof(T)))
121  {
122  stringstream error;
123  error << "[send] with [buffer=" << data << "] [len=" << len << "] Data length higher then max buffer length";
124  throw SocketException(error.str());
125  }
126 
127  int received_bytes;
128  socklen_t size = sizeof(struct sockaddr);
129 
130  if ((received_bytes = recvfrom(this->_socket_id, (char*)data, len, 0, (struct sockaddr*)address, (socklen_t*)&size)) == -1)
131  {
132  throw SocketException("[receive] Cannot receive");
133  }
134 
135  *received_elements = (unsigned int)(received_bytes / sizeof(T));
136 
137  return received_bytes;
138  }
