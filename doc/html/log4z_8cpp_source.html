529 static void trimLogConfig(std::string &str, std::string extIgnore)
530 {
531  if (str.empty()){return;}
532  extIgnore += "\r\n\t ";
533  int length = (int)str.length();
534  int posBegin = 0;
535  int posEnd = 0;
536 
537  //trim utf8 file bom
538  if (str.length() >= 3
539  && (unsigned char)str[0] == 0xef
540  && (unsigned char)str[1] == 0xbb
541  && (unsigned char)str[2] == 0xbf)
542  {
543  posBegin = 3;
544  }
545 
546  //trim character
547  for (int i = posBegin; i<length; i++)
548  {
549  bool bCheck = false;
550  for (int j = 0; j < (int)extIgnore.length(); j++)
551  {
552  if (str[i] == extIgnore[j])
553  {
554  bCheck = true;
555  }
556  }
557  if (bCheck)
558  {
559  if (i == posBegin)
560  {
561  posBegin++;
562  }
563  }
564  else
565  {
566  posEnd = i + 1;
567  }
568  }
569  if (posBegin < posEnd)
570  {
571  str = str.substr(posBegin, posEnd-posBegin);
572  }
573  else
574  {
575  str.clear();
576  }
577 }

591 {

601  {
602  return true;
603  }
604 
605  if (kv.first.at(0) == '[')
606  {
607  trimLogConfig(kv.first, "[]");
608  key = kv.first;
609  {
610  std::string tmpstr = kv.first;
611  std::transform(tmpstr.begin(), tmpstr.end(), tmpstr.begin(), ::tolower);
612  if (tmpstr == "main")
613  {
614  key = "Main";
615  }

623  li._name = key;
