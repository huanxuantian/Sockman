379 #define LOG_STREAM(id, level, file, line, log)\
380 do{\
381  if (zsummer::log4z::ILog4zManager::getPtr()->prePushLog(id,level)) \
382  {\
383  zsummer::log4z::LogData * pLog = zsummer::log4z::ILog4zManager::getPtr()->makeLogData(id, level); \
384  zsummer::log4z::Log4zStream ss(pLog->_content + pLog->_contentLen, LOG4Z_LOG_BUF_SIZE - pLog->_contentLen);\
385  ss << log;\
386  pLog->_contentLen += ss.getCurrentLen(); \
387  zsummer::log4z::ILog4zManager::getPtr()->pushLog(pLog, file, line);\
388  }\
389 } while (0)
